// ===== DATES ===== const start = new Date("2026-04-26"); // ===== ROUTE ===== const route = [ { name: "Shanghai 🇨🇳", distance: 0 }, { name: "Singapore 🌴", distance: 4300 }, { name: "Indian Ocean 🌊", distance: 11000 }, { name: "Suez Canal 🐪", distance: 14500 }, { name: "Europe 🇪🇺", distance: 20000 } ]; // use LET (important!) let speedPerDay = 750; // ===== UPDATE ===== function update() { const now = new Date(); const daysPassed = (now - start) / 86400000; const totalDist = route[route.length - 1].distance; let traveled = daysPassed * speedPerDay; if (traveled < 0) traveled = 0; if (traveled > totalDist) traveled = totalDist; const progress = (traveled / totalDist) * 100; // ===== STATUS ===== let status = route[route.length - 1].name; for (let i = 0; i < route.length - 1; i++) { if (traveled >= route[i].distance && traveled < route[i + 1].distance) { status = route[i + 1].name; break; } } const remaining = totalDist - traveled; function daysTo(target) { return Math.max(0, Math.ceil((target - traveled) / speedPerDay)); } const suezDist = route[3].distance; // ===== UI ===== document.getElementById("bar").style.width = progress + "%"; document.getElementById("percent").innerText = "Progress: " + progress.toFixed(1) + "%"; document.getElementById("distance").innerText = "Remaining: " + Math.round(remaining) + " km"; document.getElementById("speed").innerText = "Speed: ~17 knots ⚡"; document.getElementById("status").innerText = status; document.getElementById("suez").innerText = "Suez: " + daysTo(suezDist) + " days"; document.getElementById("arrival").innerText = "Arrival: " + daysTo(totalDist) + " days"; // ===== ALERTS ===== if (daysTo(suezDist) <= 1 && !localStorage.getItem("alert1")) { showBanner("🚢 Suez coming up!"); localStorage.setItem("alert1", "1"); } if (progress > 95 && !localStorage.getItem("alert2")) { showBanner("🇪🇺 Almost arrived!"); localStorage.setItem("alert2", "1"); } }
